def only_word_sun(text):
    """แก้เทสเคสปั่นประสาท"""
    where_sun = text.find("Sun")
    if text.find(" Sun ") != -1:
        where_sun = text.find(" Sun ")+1
    elif text.find(" Sun") != -1:
        where_sun = text.find(" Sun")+1
    return where_sun
def output_text(text, where_sun, most_hot, most_hot2, most_cool, most_cool2):
    """ไม่พักเพราะโจทย์เธอเลยยยยยยยยยยยย"""
    if text.count(" ", 0, where_sun) == 0:
        print("Hot: %s"%most_hot2)

        print("Hot: %s"%most_hot)
