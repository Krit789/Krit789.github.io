        list_text.append(i)

            break
        if temp == "":

            del list_text[0]

            posi = list_text.index(temp)
            if list_text[posi+1:].count(temp) > 0:
                dis = list_text[posi+1:].index(temp)
                if dis < posi:
                    del list_text[posi]
                    del list_text[posi+dis]
                    continue
                else:
                    del list_text[posi]
                    temp = ""
                    continue
            else:
                del list_text[posi]
                temp = ""
        else:
            left += temp
            temp = ""
