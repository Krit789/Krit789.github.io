def temp_maker(text, temp):
    for i, j in enumerate(text):
        if i > 1 and text[i-2] + text[i-1] + j == 'Sun':
            temp += str(i+1) + ' '
        elif j == ' ' and text[i-3] + text[i-2] + text[i-1] != 'Sun':
            temp += str(i) + ' '
    return temp
