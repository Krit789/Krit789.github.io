    """find"""
    if solar.count(planet) == 1:

        return posi

def amount(solar):
    """amount"""
    total = 0
    text = ""
    check = 0
    for i in solar:
        if i.isupper() == True:
            total += 1
            text += i
            check = 1
        elif check == 1:

            check = 0
    return total, text
