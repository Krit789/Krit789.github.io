def order_time(num1, num2, num3, num4, dire):
    """compare 2 thing"""
    if num4 >= num3 and num4 > -1 and num3 > -1: num4, num3 = num3, num4

def get_info(reply, rip):
    """extraction function"""
    if reply.count("Reply from " + rip) > 0 and reply[reply.index(" time") + 5] == "=":
        complete = True
        time = reply[reply.index(" time") + 6:reply.index("ms")]
        time = int(time)
    elif reply.count("Reply from " + rip) > 0 and reply[reply.index(" time") + 5] == "<":
        complete = True
        time = reply[reply.index(" time") + 6:reply.index("ms")]
        time = int(time) - 1
    else:
        complete = False
        time = -1
    return complete, time
